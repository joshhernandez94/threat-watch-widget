const vendors = [ "Microsoft", "Lenovo", "SonicWall", "Dell", "Meraki", "Cisco", "Aruba", "HPE" ]; const feedEl = document.getElementById("feed"); fetch("https://api.rss2json.com/v1/api.json?rss_url=https://www.exploit-db.com/rss.xml") .then(res => res.json()) .then(data => { const items = data.items.filter(item => vendors.some(v => item.title.toLowerCase().includes(v.toLowerCase()) ) ).slice(0,6); if(items.length === 0){ feedEl.innerHTML = "No matching vendor exploits found."; return; } feedEl.innerHTML = items.map(item => `
Exploit
${item.title}
Exploit-DB • ${new Date(item.pubDate).toLocaleDateString()}
`).join(""); }) .catch(() => { feedEl.innerHTML = "Could not load threat feed."; });
